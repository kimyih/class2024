01. 함수 : 선언적 함수


선언적 함수는 구현 세부사항을 숨기고 목적을 명확히 전달하는 함수 스타일입니다.
이러한 방식은 추상화된 인터페이스를 제공하여 코드의 가독성과 유지보수성을 향상시킵니다.
함수형 프로그래밍의 중요한 개념 중 하나로, 코드를 이해하기 쉽고 변경하기 용이하게 합니다. 

{
    function func() {
        console.log("01. 함수가 실행되었습니다.");
    }
    func();
}

02. 함수 : 익명 함수


익명 함수는 이름이 없는 함수로, 주로 변수에 할당되거나 다른 함수의 인자로 전달되어 사용됩니다.
일급 객체로 취급되어 변수에 할당하거나 콜백 함수로 활용되며, 스코프 내에서만 영향을 미칩니다.
함수 표현식을 사용하여 정의되며, 코드의 모듈화와 스코프 충돌을 방지하는 데 유용합니다. 

{
    const func = function () {
        console.log("02. 함수가 실행되었습니다.");
    }
    func();
}

03. 함수 : 매개변수 함수


매개변수 함수는 다른 함수의 매개변수로 전달되는 함수를 가리킵니다.
이는 주로 콜백 함수로 활용되며, 함수의 동작을 확장하거나 조절하는 데 사용됩니다.
매개변수 함수를 이용하면 코드의 재사용성과 유연성을 높일 수 있으며, 함수형 프로그래밍에서는 코드를 더욱 간결하고 추상화된 수준으로 작성할 수 있습니다. 

{
   // 선언적 함수
    function func(str) {
        console.log(str);
    }
    func("03. 함수가 실행되었습니다.");

    //익명 함수
    const func1 = function (str) {
        console.log(str);
    }
    func("03. 함수가 실행되었습니다.");
}

04. 함수 : 리턴값 함수


리턴값 함수는 함수가 실행 후 반환하는 값을 가리킵니다.
이 반환값은 함수가 호출된 코드로 전달되어 활용됩니다.
반환값은 함수의 결과나 다음 단계의 처리에 활용될 수 있습니다. 

{
    //선언적 함수
    function func() {
        return "04. 함수가 실행되었습니다."
    }
    console.log(func());

    //익명 함수
    const func1 = function () {
        return "04. 함수가 실행되었습니다."
    }
    console.log(func1());
}

05. 함수 : 매개변수 + 리턴값 함수


매개변수와 리턴값을 동시에 갖는 함수는 외부로부터 입력값을 받아 처리하고, 그 결과를 반환합니다.
이 함수는 입력값을 기반으로 연산을 수행하고, 그 결과를 호출한 코드로 반환하여 활용됩니다.
매개변수와 반환값을 활용하여 프로그램의 동작을 제어하고 다양한 작업을 수행할 수 있습니다. 

{
    // 선언적 함수
    function func(str) {
        return str;
    }
    console.log(func("05. 함수가 실행되었습니다."));

    // 익명 함수
    const func1 = function (str) {
        return str;
    }
    console.log(func1("05. 함수가 실행되었습니다."));
}

06. 화살표 함수 : 선언적 함수


화살표 함수는 ES6에서 도입된 간결한 함수 정의 방식으로, 일반 함수보다 구문이 간결하며 주로 익명 함수로 사용됩니다.
외부 스코프의 this를 그대로 사용하므로 this 바인딩에 대한 혼란을 줄일 수 있습니다.
한 줄의 표현식에서는 중괄호와 return 문을 생략하고 자동으로 반환됩니다. 

{
    // 선언적 함수 
    function func() {
        console.log("06. 함수가 실행되었습니다.")
    }
    func();

    // 선언적 함수를 화살표 함수로 바꿈
    func = () => {
        console.log("06. 함수가 실행되었습니다.")
    }
    func();
}

07. 화살표 함수 : 익명 함수


화살표 함수는 ES6에서 소개된 간결한 함수 정의 방식으로, 익명 함수를 생성할 때 주로 활용됩니다.
함수 표현식보다 구문이 간결하며, => 기호를 사용하여 정의되며, 주로 콜백 함수나 간단한 함수 표현식에서 사용됩니다. 

{

    const func1 = () => {
        console.log("07. 함수가 실행되었습니다.")
    };
    func1();
}

08. 화살표 함수 : 매개변수 함수


화살표 함수를 매개변수 함수로 사용하면 다른 함수의 매개변수에 간결한 함수 정의를 전달할 수 있습니다.
이는 주로 콜백 함수로 활용되며, 함수 호출 시 동작을 조정하거나 특정 기능을 수행하기 위해 활용됩니다.
매개변수 함수로 전달된 화살표 함수는 해당 함수 내에서 호출되어 특정 작업을 수행하고 결과를 반환하거나 다른 동작을 수행할 수 있습니다. 

{
    // 선언적 함수 
    func = (str) => {
        console.log(str);
    }
    func("08. 함수가 실행되었습니다.");
        
    // 익명 함수
    const func1 = (str) => {
        console.log(str);
    }
    func1("08. 함수가 실행되었습니다.");
}

09. 화살표 함수 : 리턴값 함수(결과, 종료)


화살표 함수를 이용한 리턴값 함수는 함수 내에서 특정 조건에 따라 값을 반환하거나 함수의 실행을 종료시키는 역할을 합니다.
주로 다른 함수 내에서 사용되며, 간결한 구문으로 정의되어 코드의 가독성을 높이고 프로그래밍의 효율성을 증진시킵니다.
한 줄로 표현되는 경우에는 중괄호와 return 문을 생략할 수 있어 코드를 더욱 간결하게 작성할 수 있습니다. 

{
    //선언적 함수

    //익명 함수

        return "09. 함수가 실행되었습니다."
    }
    console.log(func1());
}

10. 화살표 함수 : 매개변수 + 리턴값 함수


화살표 함수를 이용한 매개변수와 리턴값을 동시에 갖는 함수는 입력값을 받아 처리한 후 결과를 반환합니다.
주로 다른 함수의 매개변수로 전달되어 사용되며, 간결한 문법으로 함수를 정의하고 실행할 수 있습니다.
이를 통해 코드를 간결하게 작성하고 가독성을 높일 수 있습니다. 

{
    // 익명 함수(화살표 함수) + 매개변수 + 리턴값
    const func = (str) => {
        return str;
    }
    console.log(func("10. 함수가 실행되었습니다."));

    // 매개변수가 하나이면 괄호 생략 가능
    const func2 = str => {
        return str;
    }
    console.log(func2("10. 함수가 실행되었습니다."));

    // 리턴 생략
    const func3 = str => str;
    console.log(func3("10. 함수가 실행되었습니다."));

    // 선언적 함수(가독성X)
    func4 = str => str;
    console.log(func4("10. 함수가 실행되었습니다."));
}
